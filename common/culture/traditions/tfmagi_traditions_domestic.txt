	category = regional
	layers = {
		0 = diplo
		1 = mediterranean
		4 = greeting.dds
	}

		always = yes
	}

	can_pick = {
		PREV = {
			OR = {
				any_vassal_or_below = {
					count >= 3

				}

					culture.has_cultural_tradition = tradition_tfmagi_cosmopolitan_tolerance
				}
			}
		}
	}

		tfmagi_can_undertake_minority_investments = yes
		cheaper_to_hybridize = yes
		easier_to_hybridize = yes
	}

	character_modifier = {
		general_opinion = 10
		ignore_different_faith_opinion = yes
	}

	ai_will_do = {
		value = 400
	}
}
